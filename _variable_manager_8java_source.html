102  public static String replaceVars(String format, UUID uuid, String msg, Boolean doColour) {
103  NavigableMap<String, String> fVarMap = new TreeMap<String, String>();
104  NavigableMap<String, String> nVarMap = new TreeMap<String, String>();
105  NavigableMap<String, String> lVarMap = new TreeMap<String, String>();
106 
107  for (Var var : varSet) {
108  Method[] methods = var.getClass().getMethods();
109 
110  for (Method method : methods) {
111  ResolvePriority priority = ResolvePriority.DEFAULT;
112  IndicatorType type = IndicatorType.MISC_VAR;
113  String[] keys = {};
114  String value = "";
115 
116  if (method.isAnnotationPresent(Var.Keys.class)) {
117  Var.Keys vKeys = method.getAnnotation(Var.Keys.class);
118 
119  keys = vKeys.keys();
120 
121  if (msg != null && var instanceof MessageVars.MessageVar) {
122  value = msg;
123  } else {
124  value = var.getValue(uuid);
125  }
126  }
127 
128  if (method.isAnnotationPresent(Var.Meta.class)) {
129  Var.Meta vMeta = method.getAnnotation(Var.Meta.class);
130 
131  priority = vMeta.priority();
132  type = vMeta.type();
133  }
134 
135  switch (priority) {
136  case FIRST:
137  for (String key : keys) {
138  fVarMap.put(type.getValue() + key, value);
139  }
140 
141  break;
142  case DEFAULT:
143  for (String key : keys) {
144  nVarMap.put(type.getValue() + key, value);
145  }
146 
147  break;
148  case LAST:
149  for (String key : keys) {
150  lVarMap.put(type.getValue() + key, value);
151  }
152 
153  break;
154  default:
155  for (String key : keys) {
156  nVarMap.put(type + key, value);
157  }
158 
159  break;
160  }
161  }
162  }
163 
164  fVarMap = fVarMap.descendingMap();
165  nVarMap = nVarMap.descendingMap();
166  lVarMap = lVarMap.descendingMap();
167 
168  format = replacer(format, fVarMap, doColour);
