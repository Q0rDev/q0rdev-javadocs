111  public static String replaceVars(String format, UUID uuid, String msg, Boolean doColour) {
112  NavigableMap<String, String> fVarMap = new TreeMap<>();
113  NavigableMap<String, String> nVarMap = new TreeMap<>();
114  NavigableMap<String, String> lVarMap = new TreeMap<>();
115 
116  for (Var var : varSet) {
117  Method[] methods = var.getClass().getMethods();
118 
119  for (Method method : methods) {
120  ResolvePriority priority = ResolvePriority.DEFAULT;
121  IndicatorType type = IndicatorType.MISC_VAR;
122  String[] keys = {};
123  String value = "";
124 
125  if (method.isAnnotationPresent(Var.Keys.class)) {
126  Var.Keys vKeys = method.getAnnotation(Var.Keys.class);
127 
128  keys = vKeys.keys();
129 
130  if (msg != null && var instanceof MessageVars.MessageVar) {
131  value = msg;
132  } else {
133  value = var.getValue(uuid);
134  }
135  }
136 
137  if (method.isAnnotationPresent(Var.Meta.class)) {
138  Var.Meta vMeta = method.getAnnotation(Var.Meta.class);
139 
140  priority = vMeta.priority();
141  type = vMeta.type();
142  }
143 
144  switch (priority) {
145  case FIRST:
146  for (String key : keys) {
147  fVarMap.put(type.getValue() + key, value);
148  }
149 
150  break;
151  case DEFAULT:
152  for (String key : keys) {
153  nVarMap.put(type.getValue() + key, value);
154  }
155 
156  break;
157  case LAST:
158  for (String key : keys) {
159  lVarMap.put(type.getValue() + key, value);
160  }
161 
162  break;
163  default:
164  for (String key : keys) {
165  nVarMap.put(type + key, value);
166  }
167 
168  break;
169  }
170  }
171  }
172 
173  fVarMap = fVarMap.descendingMap();
174  nVarMap = nVarMap.descendingMap();
175  lVarMap = lVarMap.descendingMap();
176 
177  format = replacer(format, fVarMap, doColour);
178  format = replacer(format, nVarMap, doColour);
179  format = replacer(format, lVarMap, false);
180 
181  return format;
182  }
