213  private void saveFile(File folder, String file, String u) {
214  if (!folder.exists()) {
215  folder.mkdir();
216  }
217  BufferedInputStream in = null;
218  FileOutputStream fout = null;
219  try {
220  // Download the file
221  final URL url = new URL(u);
222  final int fileLength = url.openConnection().getContentLength();
223  in = new BufferedInputStream(url.openStream());
224  fout = new FileOutputStream(folder.getAbsolutePath() + "/" + file);
225 
226  final byte[] data = new byte[Updater.BYTE_SIZE];
227  int count;
228  if (this.announce) {
229  this.plugin.getLogger().info("About to download a new update: " + this.versionName);
230  }
231  long downloaded = 0;
232  while ((count = in.read(data, 0, Updater.BYTE_SIZE)) != -1) {
233  downloaded += count;
234  fout.write(data, 0, count);
235  final int percent = (int) ((downloaded * 100) / fileLength);
236  if (this.announce && ((percent % 10) == 0)) {
237  this.plugin.getLogger().info("Downloading update: " + percent + "% of " + fileLength + " bytes.");
238  }
239  }
240  //Just a quick check to make sure we didn't leave any files from last time...
241  for (final File xFile : new File(this.plugin.getDataFolder().getParent(), this.updateFolder).listFiles()) {
242  if (xFile.getName().endsWith(".zip")) {
243  xFile.delete();
244  }
245  }
246  // Check to see if it's a zip file, if it is, unzip it.
247  final File dFile = new File(folder.getAbsolutePath() + "/" + file);
248  if (dFile.getName().endsWith(".zip")) {
249  // Unzip
250  this.unzip(dFile.getCanonicalPath());
251  }
252  if (this.announce) {
253  this.plugin.getLogger().info("Finished updating.");
254  }
255  } catch (final Exception ex) {
256  this.plugin.getLogger().warning("The auto-updater tried to download a new update, but was unsuccessful.");
257  this.result = Updater.UpdateResult.FAIL_DOWNLOAD;
258  } finally {
259  try {
260  if (in != null) {
261  in.close();
262  }
263  if (fout != null) {
264  fout.close();
265  }
266  } catch (final Exception ex) {
267  }
268  }
269  }

274  private void unzip(String file) {
275  try {
276  final File fSourceZip = new File(file);
277  final String zipPath = file.substring(0, file.length() - 4);
278  ZipFile zipFile = new ZipFile(fSourceZip);
279  Enumeration<? extends ZipEntry> e = zipFile.entries();
280  while (e.hasMoreElements()) {
281  ZipEntry entry = e.nextElement();
282  File destinationFilePath = new File(zipPath, entry.getName());
283  destinationFilePath.getParentFile().mkdirs();
284  if (entry.isDirectory()) {
285  continue;
286  } else {
287  final BufferedInputStream bis = new BufferedInputStream(zipFile.getInputStream(entry));
288  int b;
289  final byte buffer[] = new byte[Updater.BYTE_SIZE];
290  final FileOutputStream fos = new FileOutputStream(destinationFilePath);
291  final BufferedOutputStream bos = new BufferedOutputStream(fos, Updater.BYTE_SIZE);
292  while ((b = bis.read(buffer, 0, Updater.BYTE_SIZE)) != -1) {
293  bos.write(buffer, 0, b);
294  }
295  bos.flush();
296  bos.close();
297  bis.close();
298  final String name = destinationFilePath.getName();
299  if (name.endsWith(".jar") && this.pluginFile(name)) {
300  destinationFilePath.renameTo(new File(this.plugin.getDataFolder().getParent(), this.updateFolder + "/" + name));
301  }
302  }
303  entry = null;
304  destinationFilePath = null;
305  }
306  e = null;
307  zipFile.close();
308  zipFile = null;
309 
310  // Move any plugin data folders that were included to the right place, Bukkit won't do this for us.
311  for (final File dFile : new File(zipPath).listFiles()) {
312  if (dFile.isDirectory()) {
313  if (this.pluginFile(dFile.getName())) {
314  final File oFile = new File(this.plugin.getDataFolder().getParent(), dFile.getName()); // Get current dir
315  final File[] contents = oFile.listFiles(); // List of existing files in the current dir
316  for (final File cFile : dFile.listFiles()) // Loop through all the files in the new dir
317  {
318  boolean found = false;
319  for (final File xFile : contents) // Loop through contents to see if it exists
320  {
321  if (xFile.getName().equals(cFile.getName())) {
322  found = true;
323  break;
324  }
325  }
326  if (!found) {
327  // Move the new file into the current dir
328  cFile.renameTo(new File(oFile.getCanonicalFile() + "/" + cFile.getName()));
329  } else {
330  // This file already exists, so we don't need it anymore.
331  cFile.delete();
332  }
333  }
334  }
335  }
336  dFile.delete();
337  }
338  new File(zipPath).delete();
339  fSourceZip.delete();
340  } catch (final IOException ex) {
341  this.plugin.getLogger().warning("The auto-updater tried to unzip a new update file, but was unsuccessful.");
342  this.result = Updater.UpdateResult.FAIL_DOWNLOAD;
343  ex.printStackTrace();
344  }
345  new File(file).delete();
346  }

363  private boolean versionCheck(String title) {
364  if (this.type != UpdateType.NO_VERSION_CHECK) {
365  Integer length = plugin.getDescription().getVersion().split("-").length;
366  String version = plugin.getDescription().getVersion().split("-")[length - 1].split("git")[0];
367 
368  if (title.split(" ").length == 2) {
369  final String remoteVersion = title.split(" ")[1]; // Get the newest file's version number
370 
371  if (this.hasTag(version) || version.equalsIgnoreCase(remoteVersion)) {
372  // We already have the latest version, or this build is tagged for no-update
373  this.result = Updater.UpdateResult.NO_UPDATE;
374  return false;
375  }
376  } else {
377  // The file's name did not contain the string 'vVersion'
378  final String authorInfo = this.plugin.getDescription().getAuthors().size() == 0 ? "" : " (" + this.plugin.getDescription().getAuthors().get(0) + ")";
379  this.plugin.getLogger().warning("The author of this plugin" + authorInfo + " has misconfigured their Auto Update system");
380  this.plugin.getLogger().warning("File versions should follow the format 'PluginName vVERSION'");
381  this.plugin.getLogger().warning("Please notify the author of this error.");
382  this.result = Updater.UpdateResult.FAIL_NOVERSION;
383  return false;
384  }
385  }
386 
387  return true;
388  }

402  private boolean read() {
403  try {
404  final URLConnection conn = this.url.openConnection();
405  conn.setConnectTimeout(5000);
406 
407  if (this.apiKey != null) {
408  conn.addRequestProperty("X-API-Key", this.apiKey);
409  }
410  conn.addRequestProperty("User-Agent", "Updater (by Gravity)");
411 
412  conn.setDoOutput(true);
413 
414  final BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream()));
415  final String response = reader.readLine();
416 
417  final JSONArray array = (JSONArray) JSONValue.parse(response);
418 
419  if (array.size() == 0) {
420  this.plugin.getLogger().warning("The updater could not find any files for the project id " + this.id);
421  this.result = UpdateResult.FAIL_BADID;
422  return false;
423  }
424 
425  this.versionName = (String) ((JSONObject) array.get(array.size() - 1)).get(Updater.TITLE_VALUE);
426  this.versionLink = (String) ((JSONObject) array.get(array.size() - 1)).get(Updater.LINK_VALUE);
427  this.versionType = (String) ((JSONObject) array.get(array.size() - 1)).get(Updater.TYPE_VALUE);
428  this.versionGameVersion = (String) ((JSONObject) array.get(array.size() - 1)).get(Updater.VERSION_VALUE);
429 
430  return true;
431  } catch (final IOException e) {
432  if (e.getMessage().contains("HTTP response code: 403")) {
433  this.plugin.getLogger().warning("dev.bukkit.org rejected the API key provided in plugins/Updater/config.yml");
434  this.plugin.getLogger().warning("Please double-check your configuration to ensure it is correct.");
435  this.result = UpdateResult.FAIL_APIKEY;
436  } else {
437  this.plugin.getLogger().warning("The updater could not contact dev.bukkit.org for updating.");
438  this.plugin.getLogger().warning("If you have not recently modified your configuration and this is the first time you are seeing this message, the site may be experiencing temporary downtime.");
439  this.result = UpdateResult.FAIL_DBO;
440  }
441  e.printStackTrace();
442  return false;
443  }
444  }

446  private class UpdateRunnable implements Runnable {
447 
448  @Override
449  public void run() {
450  if (Updater.this.url != null) {
451  // Obtain the results of the project's file feed
452  if (Updater.this.read()) {
453  if (Updater.this.versionCheck(Updater.this.versionName)) {
454  if ((Updater.this.versionLink != null) && (Updater.this.type != UpdateType.NO_DOWNLOAD)) {
455  String name = Updater.this.file.getName();
456  // If it's a zip file, it shouldn't be downloaded as the plugin's name
457  if (Updater.this.versionLink.endsWith(".zip")) {
458  final String[] split = Updater.this.versionLink.split("/");
459  name = split[split.length - 1];
460  }
461  Updater.this.saveFile(new File(Updater.this.plugin.getDataFolder().getParent(), Updater.this.updateFolder), name, Updater.this.versionLink);
462  } else {
463  Updater.this.result = UpdateResult.UPDATE_AVAILABLE;
464  }
465  }
466  }
467  }
